## 2. 构建系统设计

### 2.1 顶层 CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.25)
project(qoocore VERSION 0.1.0 LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

# ---- 编译选项 ----
option(QOOCORE_BUILD_PYTHON  "Build Python bindings"      ON)
option(QOOCORE_BUILD_CLI     "Build CLI tool"             ON)
option(QOOCORE_BUILD_TESTS   "Build tests"                ON)
option(QOOCORE_BUILD_SAMPLES "Build sample programs"      ON)

option(QOOCORE_ENABLE_NPU_QNN  "Enable Qualcomm QNN"    ON)
option(QOOCORE_ENABLE_NPU_BPU  "Enable Horizon BPU"      OFF)
option(QOOCORE_ENABLE_NPU_RKNN "Enable Rockchip RKNN"    OFF)
option(QOOCORE_ENABLE_GPU_CUDA "Enable NVIDIA CUDA"      ON)
option(QOOCORE_ENABLE_GPU_OPENCL "Enable OpenCL"         OFF)
option(QOOCORE_ENABLE_DSP       "Enable DSP"             OFF)

# ---- 依赖 ----
find_package(FlatBuffers REQUIRED)
find_package(spdlog REQUIRED)
find_package(yaml-cpp REQUIRED)
find_package(ONNXRuntime REQUIRED)

if(QOOCORE_ENABLE_NPU_QNN)
    find_package(QNN REQUIRED)
endif()

if(QOOCORE_ENABLE_GPU_CUDA)
    find_package(CUDA REQUIRED)
    enable_language(CUDA)
endif()

# ---- 子目录 ----
add_subdirectory(src/core)
add_subdirectory(src/engine)
add_subdirectory(src/compiler)
add_subdirectory(src/runtime)
add_subdirectory(src/operators)
add_subdirectory(src/memory)
add_subdirectory(src/hardware)
add_subdirectory(src/profiler)
add_subdirectory(src/cloud)

if(QOOCORE_BUILD_CLI)
    add_subdirectory(cli)
endif()

if(QOOCORE_BUILD_PYTHON)
    add_subdirectory(python)
endif()

if(QOOCORE_BUILD_TESTS)
    enable_testing()
    add_subdirectory(tests)
endif()

# ---- 安装 ----
install(DIRECTORY include/qoocore DESTINATION include)
install(FILES cmake/QooCoreConfig.cmake DESTINATION lib/cmake/QooCore)
```

### 2.2 构建命令

```bash
# 配置（Linux aarch64，目标 NPU + GPU）
mkdir build && cd build
cmake .. \
  -DCMAKE_BUILD_TYPE=Release \
  -DQOOCORE_ENABLE_NPU_QNN=ON \
  -DQOOCORE_ENABLE_GPU_CUDA=ON \
  -DQOOCORE_BUILD_PYTHON=ON \
  -DQOOCORE_BUILD_TESTS=ON

# 编译（并行）
make -j$(nproc)

# 安装到系统
sudo make install

# Python 包安装
pip install python/

# 运行测试
ctest --output-on-failure
```

## 3. 代码规范

### 3.1 命名规范

| 类型 | 规范 | 示例 |
|------|------|------|
| 类名 | PascalCase | `InferenceEngine` |
| 函数名 | snake_case | `load_model()` |
| 变量名 | snake_case | `model_handle` |
| 成员变量 | snake_case + trailing underscore | `backend_type_` |
| 常量 | UPPER_SNAKE_CASE | `MAX_MODELS` |
| 命名空间 | lowercase | `qoocore` |
| 文件名（实现） | snake_case | `inference_engine.cpp` |
| 文件名（头文件） | snake_case | `inference_engine.h` |
| 枚举值 | PascalCase 或 UPPER_SNAKE | `BackendType::NPU` |

### 3.2 .clang-format 配置

```yaml
BasedOnStyle: Google
IndentWidth: 4
ColumnLimit: 100
PointerAlignment: Left
ReferenceAlignment: Left
AllowShortFunctionsOnASingleLine: None
```

### 3.3 目录组织原则

1. 每个子系统一个目录（`compiler/`, `runtime/`, ...）
2. 头文件在 `include/qoocore/`，实现在 `src/<subsystem>/`
3. 测试文件与源文件对应（`test_xxx.cpp` 在 `tests/unit/`）
4. 第三方依赖统一在 `third_party/`（git submodule）
5. Schema 文件统一在 `schema/`（`.fbs`, `.proto`）

## 5. 初始化骨架脚本

首次搭建项目时，运行以下脚本创建目录结构（Phase 1 基础设施）：

```bash
#!/bin/bash
# scripts/init_qoocore.sh - 初始化 qoocore 目录结构

ROOT="qoocore"

mkdir -p "$ROOT"/{include/qoocore/hal,src/{core,engine,compiler/{importer,ir,optimizer,quantizer,pruner,codegen},runtime/{engine,hal,backend,scheduler,pipeline},operators/{vision,pointcloud,bev,trajectory,signal,multimodal},memory/{pool,compressor,swapper,ion},hardware/{probe,powermgr,peripheral,secure},profiler/{latency,accuracy,resource,power,diagnostic,ui},cloud/{sync,hybrid,federated,ota}},python/qoocore,cli/commands,schema,tests/{unit,integration,benchmark,data/{models,images}},third_party,cmake}

# 创建占位头文件（避免 CMake 报错）
touch "$ROOT/include/qoocore/"{core.h,engine.h,tensor.h,model.h,backend.h,compiler.h,config.h}
touch "$ROOT/include/qoocore/hal/npu_hal.h"
touch "$ROOT/src/core/"{result.cpp,dtype.cpp}
touch "$ROOT/CMakeLists.txt"
touch "$ROOT/.clang-format"
touch "$ROOT/package.xml"

echo "qoocore 目录结构初始化完成"
```
