## 2. 构建配置

### 2.1 顶层 CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.22)
project(qoosvc VERSION 0.1.0 LANGUAGES CXX)

# C++17 标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 编译选项
add_compile_options(-Wall -Wextra -Wpedantic)

# 子目录
add_subdirectory(src/svc_manager)
add_subdirectory(src/svc_voice)
add_subdirectory(src/svc_navigation)
add_subdirectory(src/svc_spatial)
add_subdirectory(src/svc_diagnostics)
add_subdirectory(src/svc_hmi)
add_subdirectory(src/svc_charging)
add_subdirectory(src/svc_people)
add_subdirectory(src/svc_multi_robot)

# 公共库
add_library(qoosvc_common INTERFACE)
target_include_directories(qoosvc_common INTERFACE
    ${CMAKE_SOURCE_DIR}/include
)

# 测试
enable_testing()
add_subdirectory(test)
```

| `python/qoosvc/` | Python | Python API 封装 + CLI + AI/诊断逻辑 | pybind11, Typer, Rich |

## 4. 命名规范

### 4.1 C++ 命名规范

| 类型 | 规范 | 示例 |
|------|------|------|
| 命名空间 | snake_case | `qoosvc::voice` |
| 类/结构体 | PascalCase | `VoiceService`, `ASRResult` |
| 函数/方法 | snake_case | `recognize()`, `set_wake_word()` |
| 成员变量 | snake_case_ (尾下划线) | `wake_word_`, `is_active_` |
| 常量 | kPascalCase | `kMaxRetryAttempts` |
| 宏 | UPPER_SNAKE_CASE | `QOOSVC_VERSION` |
| 文件名 | snake_case | `voice_service.cpp`, `astar_planner.h` |

### 4.2 Python 命名规范

| 类型 | 规范 | 示例 |
|------|------|------|
| 模块 | snake_case | `voice_service.py` |
| 类 | PascalCase | `VoiceService`, `ASRResult` |
| 函数/方法 | snake_case | `recognize()`, `set_wake_word()` |
| 变量 | snake_case | `wake_word`, `is_active` |
| 常量 | UPPER_SNAKE_CASE | `MAX_RETRY_ATTEMPTS` |

## 5. 构建与运行

### 5.1 本地构建

```bash
# C++ 模块构建
mkdir build && cd build
cmake .. -DCMAKE_BUILD_TYPE=Release
make -j$(nproc)

# Python 包安装
pip install -e .

# 完整构建
./scripts/build_all.sh
```

### 5.2 运行服务

```bash
# 启动所有系统服务
qoo-svc start --all

# 启动指定服务
qoo-svc start svc_voice svc_navigation

# 查看服务状态
qoo-svc status

# 运行诊断
qoo-svc diag post
qoo-svc diag report
```

### 5.3 机器人端 Docker 构建（运行于机器人计算平台）

```dockerfile
# deploy/docker/Dockerfile.qoosvc
# 注意：qoosvc 是机器人端服务，运行在机器人计算平台上，非云端容器
FROM nvidia/cuda:12.1-runtime-ubuntu22.04

RUN apt-get update && apt-get install -y \
    libonnxruntime1.17 \
    libeigen3-dev \
    python3.11 \
    && rm -rf /var/lib/apt/lists/*

COPY build/lib/ /usr/local/lib/
COPY python/ /opt/qoosvc/python/
COPY config/ /etc/qoosvc/

ENV LD_LIBRARY_PATH=/usr/local/lib
ENV PYTHONPATH=/opt/qoosvc

ENTRYPOINT ["python3", "-m", "qoosvc.cli.main", "start", "--all"]
```
