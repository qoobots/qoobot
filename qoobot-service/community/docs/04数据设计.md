## 二、数据库表设计 (PostgreSQL 16)

### 2.1 用户扩展表

```sql
-- 用户资料扩展 (关联 qooauth users)
CREATE TABLE user_profiles (
    id          BIGSERIAL PRIMARY KEY,
    user_id     VARCHAR(64)  NOT NULL UNIQUE,     -- qooauth user_id
    nickname    VARCHAR(100) NOT NULL,
    avatar_url  VARCHAR(512),
    bio         TEXT,
    company     VARCHAR(200),
    title       VARCHAR(200),
    location    VARCHAR(200),
    website     VARCHAR(512),
    github      VARCHAR(200),
    reputation  INT DEFAULT 0,                     -- 社区声望
    created_at  TIMESTAMP DEFAULT NOW(),
    updated_at  TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_profiles_user_id ON user_profiles(user_id);
CREATE INDEX idx_profiles_reputation ON user_profiles(reputation DESC);
```

### 2.2 贡献者表

```sql
CREATE TABLE contributors (
    id              BIGSERIAL PRIMARY KEY,
    user_id         VARCHAR(64) NOT NULL UNIQUE,
    cla_signed      BOOLEAN DEFAULT FALSE,
    cla_signed_at   TIMESTAMP,
    cla_type        VARCHAR(20),                   -- INDIVIDUAL / CORPORATE
    level           VARCHAR(20) DEFAULT 'CONTRIBUTOR', -- CONTRIBUTOR/MAINTAINER/COMMITTER/TSC
    pr_count        INT DEFAULT 0,
    commit_count    INT DEFAULT 0,
    review_count    INT DEFAULT 0,
    active_months   INT DEFAULT 0,
    joined_at       TIMESTAMP DEFAULT NOW(),
    promoted_at     TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_contributors_user ON contributors(user_id);
CREATE INDEX idx_contributors_level ON contributors(level);
```

### 2.3 论坛表

```sql
CREATE TABLE forum_categories (
    id          BIGSERIAL PRIMARY KEY,
    name        VARCHAR(100) NOT NULL,
    slug        VARCHAR(100) NOT NULL UNIQUE,
    description TEXT,
    sort_order  INT DEFAULT 0,
    parent_id   BIGINT REFERENCES forum_categories(id),
    topic_count INT DEFAULT 0,
    created_at  TIMESTAMP DEFAULT NOW()
);

CREATE TABLE forum_topics (
    id              BIGSERIAL PRIMARY KEY,
    category_id     BIGINT NOT NULL REFERENCES forum_categories(id),
    user_id         VARCHAR(64) NOT NULL,
    title           VARCHAR(500) NOT NULL,
    content         TEXT NOT NULL,
    content_html    TEXT NOT NULL,
    is_pinned       BOOLEAN DEFAULT FALSE,
    is_locked       BOOLEAN DEFAULT FALSE,
    view_count      INT DEFAULT 0,
    reply_count     INT DEFAULT 0,
    like_count      INT DEFAULT 0,
    last_reply_at   TIMESTAMP,
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_topics_category ON forum_topics(category_id, created_at DESC);
CREATE INDEX idx_topics_user ON forum_topics(user_id);
-- GIN 索引用于全文搜索
ALTER TABLE forum_topics ADD COLUMN search_vector tsvector;
CREATE INDEX idx_topics_search ON forum_topics USING GIN(search_vector);

CREATE TABLE forum_replies (
    id          BIGSERIAL PRIMARY KEY,
    topic_id    BIGINT NOT NULL REFERENCES forum_topics(id) ON DELETE CASCADE,
    user_id     VARCHAR(64) NOT NULL,
    parent_id   BIGINT REFERENCES forum_replies(id),
    content     TEXT NOT NULL,
    content_html TEXT NOT NULL,
    like_count  INT DEFAULT 0,
    created_at  TIMESTAMP DEFAULT NOW(),
    updated_at  TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_replies_topic ON forum_replies(topic_id, created_at ASC);

CREATE TABLE forum_tags (
    id      BIGSERIAL PRIMARY KEY,
    name    VARCHAR(50) NOT NULL UNIQUE,
    slug    VARCHAR(50) NOT NULL UNIQUE,
    color   VARCHAR(7),
    topic_count INT DEFAULT 0
);

CREATE TABLE forum_topic_tags (
    topic_id BIGINT NOT NULL REFERENCES forum_topics(id) ON DELETE CASCADE,
    tag_id   BIGINT NOT NULL REFERENCES forum_tags(id) ON DELETE CASCADE,
    PRIMARY KEY (topic_id, tag_id)
);

-- 点赞/收藏
CREATE TABLE forum_likes (
    id          BIGSERIAL PRIMARY KEY,
    user_id     VARCHAR(64) NOT NULL,
    target_type VARCHAR(20) NOT NULL,           -- TOPIC / REPLY
    target_id   BIGINT NOT NULL,
    created_at  TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, target_type, target_id)
);

CREATE TABLE forum_bookmarks (
    id          BIGSERIAL PRIMARY KEY,
    user_id     VARCHAR(64) NOT NULL,
    topic_id    BIGINT NOT NULL REFERENCES forum_topics(id) ON DELETE CASCADE,
    created_at  TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, topic_id)
);
```

### 2.4 Q&A 表

```sql
CREATE TABLE qa_questions (
    id              BIGSERIAL PRIMARY KEY,
    user_id         VARCHAR(64) NOT NULL,
    title           VARCHAR(500) NOT NULL,
    content         TEXT NOT NULL,
    content_html    TEXT NOT NULL,
    view_count      INT DEFAULT 0,
    answer_count    INT DEFAULT 0,
    vote_score      INT DEFAULT 0,
    accepted_answer_id BIGINT,
    is_solved       BOOLEAN DEFAULT FALSE,
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_questions_vote ON qa_questions(vote_score DESC);
CREATE INDEX idx_questions_created ON qa_questions(created_at DESC);

ALTER TABLE qa_questions ADD COLUMN search_vector tsvector;
CREATE INDEX idx_questions_search ON qa_questions USING GIN(search_vector);

CREATE TABLE qa_answers (
    id              BIGSERIAL PRIMARY KEY,
    question_id     BIGINT NOT NULL REFERENCES qa_questions(id) ON DELETE CASCADE,
    user_id         VARCHAR(64) NOT NULL,
    content         TEXT NOT NULL,
    content_html    TEXT NOT NULL,
    vote_score      INT DEFAULT 0,
    is_accepted     BOOLEAN DEFAULT FALSE,
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_answers_question ON qa_answers(question_id, vote_score DESC);

-- 统一投票表 (问题/答案)
CREATE TABLE qa_votes (
    id              BIGSERIAL PRIMARY KEY,
    user_id         VARCHAR(64) NOT NULL,
    target_type     VARCHAR(20) NOT NULL,       -- QUESTION / ANSWER
    target_id       BIGINT NOT NULL,
    vote_type       VARCHAR(10) NOT NULL,       -- UP / DOWN
    created_at      TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, target_type, target_id)
);

CREATE TABLE qa_question_tags (
    question_id BIGINT NOT NULL REFERENCES qa_questions(id) ON DELETE CASCADE,
    tag_id      BIGINT NOT NULL REFERENCES forum_tags(id) ON DELETE CASCADE,
    PRIMARY KEY (question_id, tag_id)
);
```

### 2.5 活动表

```sql
CREATE TABLE events (
    id              BIGSERIAL PRIMARY KEY,
    title           VARCHAR(500) NOT NULL,
    slug            VARCHAR(200) NOT NULL UNIQUE,
    type            VARCHAR(30) NOT NULL,        -- DEVCON / HACKATHON / TECHTALK / WEBINAR / MEETUP
    description     TEXT,
    content_html    TEXT,
    cover_url       VARCHAR(512),
    location        VARCHAR(500),               -- 线下地址 或 ONLINE
    start_time      TIMESTAMP NOT NULL,
    end_time        TIMESTAMP NOT NULL,
    timezone        VARCHAR(50) DEFAULT 'Asia/Shanghai',
    max_attendees   INT,
    current_attendees INT DEFAULT 0,
    status          VARCHAR(20) DEFAULT 'DRAFT', -- DRAFT/PUBLISHED/ONGOING/ENDED/CANCELLED
    is_featured     BOOLEAN DEFAULT FALSE,
    created_by      VARCHAR(64) NOT NULL,
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_events_status ON events(status, start_time);
CREATE INDEX idx_events_type ON events(type, start_time);

CREATE TABLE event_registrations (
    id          BIGSERIAL PRIMARY KEY,
    event_id    BIGINT NOT NULL REFERENCES events(id) ON DELETE CASCADE,
    user_id     VARCHAR(64) NOT NULL,
    name        VARCHAR(100),
    company     VARCHAR(200),
    title       VARCHAR(200),
    email       VARCHAR(200),
    checked_in  BOOLEAN DEFAULT FALSE,
    checked_in_at TIMESTAMP,
    created_at  TIMESTAMP DEFAULT NOW(),
    UNIQUE(event_id, user_id)
);

CREATE TABLE event_agenda_items (
    id          BIGSERIAL PRIMARY KEY,
    event_id    BIGINT NOT NULL REFERENCES events(id) ON DELETE CASCADE,
    title       VARCHAR(500) NOT NULL,
    description TEXT,
    speaker     VARCHAR(200),
    speaker_title VARCHAR(200),
    start_time  TIMESTAMP NOT NULL,
    end_time    TIMESTAMP NOT NULL,
    location    VARCHAR(200),
    sort_order  INT DEFAULT 0
);

CREATE TABLE event_materials (
    id          BIGSERIAL PRIMARY KEY,
    event_id    BIGINT NOT NULL REFERENCES events(id) ON DELETE CASCADE,
    title       VARCHAR(500) NOT NULL,
    file_url    VARCHAR(512) NOT NULL,
    file_type   VARCHAR(50),                    -- PDF/PPTX/MP4/ZIP
    file_size   BIGINT,
    download_count INT DEFAULT 0,
    created_at  TIMESTAMP DEFAULT NOW()
);
```

```sql
CREATE TABLE academy_courses (

    description     TEXT,
    cover_url       VARCHAR(512),
    level           VARCHAR(20) NOT NULL,       -- BEGINNER / INTERMEDIATE / ADVANCED

    id              BIGSERIAL PRIMARY KEY,

    course_id       BIGINT NOT NULL REFERENCES academy_courses(id) ON DELETE CASCADE,

    id              BIGSERIAL PRIMARY KEY,
```

### 2.9 治理表

```sql
CREATE TABLE governance_tsc_members (
    id          BIGSERIAL PRIMARY KEY,
    user_id     VARCHAR(64) NOT NULL UNIQUE,
    role        VARCHAR(50),                    -- CHAIR / MEMBER
```

## 三、Redis 缓存设计

| Key Pattern | 类型 | 说明 | TTL |
|-------------|------|------|-----|
| `community:forum:hot` | SortedSet | 热门帖子 (score=热度) | 1h |
| `community:forum:topic:{id}` | Hash | 帖子缓存 | 30min |
| `community:qa:hot` | SortedSet | 热门问题 | 1h |
| `community:qa:question:{id}` | Hash | 问题缓存 | 30min |
| `community:event:upcoming` | List | 即将举行的活动 | 1h |
| `community:event:{id}` | Hash | 活动缓存 | 1h |
| `community:contributor:top` | SortedSet | 贡献者排行 | 24h |
| `community:session:{token}` | Hash | 用户会话 | 30min |
| `community:rate_limit:{ip}` | String | API 速率限制 | 1min |

## 六、RocketMQ 消息主题

| Topic | 生产者 | 消费者 | 说明 |
|-------|--------|--------|------|
| `community.notification` | 各微服务 | 通知服务 | 回复/点赞/活动提醒通知 |
| `community.forum.new_topic` | Community Svc | 搜索索引更新 | 新帖子索引 |
| `community.qa.new_question` | Community Svc | 搜索索引更新 | 新问题索引 |
| `community.event.reminder` | Event Svc | 通知服务 | 活动开始提醒 |
| `community.academy.progress` | Academy Svc | 分析服务 | 学习进度更新 |
| `community.contributor.update` | GitHub Webhook | Community Svc | 贡献数据同步 |
