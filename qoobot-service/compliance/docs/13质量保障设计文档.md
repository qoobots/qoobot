### 2.1 测试示例（v0.4 实际代码）

```java
// Repository 层: @DataJpaTest + H2
@Test
void shouldFindRegulationsByMarketAndCategory() {
    List<Regulation> result = repository.findByMarketAndCategory("EU", "PRIVACY_DATA");
    assertThat(result).hasSize(1);
    assertThat(result.get(0).getShortName()).isEqualTo("GDPR");
}

// Service 层: Mockito
@Test
void shouldCalculateProgressCorrectly() {
    when(itemRepo.findByChecklistId(PROJECT_ID)).thenReturn(List.of(entity1, entity2, entity3, entity4));
    ProjectProgress result = service.getProgress(PROJECT_ID);
    assertThat(result.total()).isEqualTo(4);
    assertThat(result.progressPercent()).isEqualTo(25.0);
}

// Controller 层: @WebMvcTest + MockMvc
mockMvc.perform(get("/api/v1/regulations/{market}", "EU"))
    .andExpect(status().isOk())
    .andExpect(jsonPath("$[0].id").value("GDPR"))
    .andExpect(jsonPath("$[0].status").value("ACTIVE"));
```
