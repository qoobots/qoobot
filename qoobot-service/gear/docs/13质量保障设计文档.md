## 4. SDK 测试

### 4.1 API 测试

```python
def test_accessory_manager_hotplug():
    manager = AccessoryManager()
    
    # 模拟配件连接
    sim_accessory = create_sim_gripper()
    manager.simulate_connect(sim_accessory)
    
    assert len(manager.list_accessories()) == 1
    
    gripper = manager.get_end_effector("gripper")
    assert gripper is not None
    assert gripper.grasp(50, 30) == Status.OK
    
    # 模拟断开
    manager.simulate_disconnect(sim_accessory)
    assert len(manager.list_accessories()) == 0

def test_mfq_certificate_validation():
    # 有效证书
    assert AccessoryAuthenticator.verify(valid_cert, valid_signature)
    # 过期证书
    assert not AccessoryAuthenticator.verify(expired_cert, valid_signature)
    # 签名不匹配
    assert not AccessoryAuthenticator.verify(valid_cert, wrong_signature)
```
