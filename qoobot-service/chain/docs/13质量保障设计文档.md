### 2.1 标定精度验证

```python
def test_camera_calibration_accuracy():
    """使用合成数据验证标定精度"""
    # 生成已知內参的虚拟相机
    true_params = CameraParams(fx=500, fy=500, cx=320, cy=240)
    
    # 生成虚拟标定图像
    images = generate_synthetic_images(true_params, n=50)
    
    # 运行标定
    estimated = calibrator.calibrate(images)
    
    # 验证精度
    assert abs(estimated.fx - true_params.fx) < 1.0
    assert estimated.reprojection_error < 0.1

def test_full_calibration_pipeline():
    """端到端标定流程测试"""
    result = pipeline.run_full_calibration(serial="TEST-001")
    assert result.status == "completed"
    assert result.imu.gyro_bias_stability < 10.0  # deg/h
    assert result.kinematics.end_effector_error < 2.0  # mm
```
