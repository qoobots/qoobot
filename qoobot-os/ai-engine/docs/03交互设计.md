## 2. C++ API 设计

### 2.1 核心命名空间与类型

```cpp
// === include/qoocore/core.h ===
#pragma once
#include <cstdint>
#include <string>
#include <vector>
#include <span>
#include <optional>
#include <future>

namespace qoocore {

// ============ 后端类型 ============
enum class BackendType : uint8_t {
    NPU       = 0,  // 神经网络加速芯片
    GPU       = 1,  // 图形处理器
    DSP       = 2,  // 数字信号处理器
    CPU       = 3,  // 中央处理器
    AUTO      = 255 // 自动选择（默认）
};

// ============ 数据类型 ============
enum class DType : uint8_t {
    FLOAT32 = 0,
    FLOAT16 = 1,
    INT32   = 2,
    INT8    = 3,
    INT4    = 4, // 打包存储，每字节存两个 INT4
    UINT8   = 5,
    BOOL    = 6,
};

// ============ 张量内存布局 ============
enum class TensorLayout : uint8_t {
    NCHW,   // RGB 图像常用（PyTorch 默认）
    NHWC,   // TensorFlow 默认
    NC,     // 全连接层
    NCDHW,  // 3D 卷积
};

// ============ 量化参数 ============
struct QuantParams {
    DType  target_dtype = DType::FLOAT32;
    std::vector<float> scales;      // per-tensor 或 per-channel
    std::vector<int32_t> zero_points;
    bool    per_channel = false;
};

// ============ 张量定义 ============
class Tensor {
public:
    // 构造
    Tensor() = default;
    Tensor(std::vector<int64_t> shape, DType dtype,
           TensorLayout layout = TensorLayout::NCHW,
           std::optional<QuantParams> quant = std::nullopt);

    // 从主机内存构造（拷贝数据）
    static Tensor from_host(std::vector<int64_t> shape, DType dtype,
                            const void* data, size_t size);

    // 从 ION/DMA-BUF 文件描述符构造（零拷贝）
    static Tensor from_ion_fd(int fd, std::vector<int64_t> shape,
                              DType dtype, size_t size);

    // 属性
    const std::vector<int64_t>& shape() const;
    DType dtype() const;
    TensorLayout layout() const;
    size_t size_bytes() const;
    bool is_quantized() const;
    const QuantParams& quant_params() const;

    // 数据访问
    void* mutable_data();
    const void* data() const;
    int ion_fd() const; // 若为 ION 内存，返回 fd；否则返回 -1

    // 维度操作
    Tensor reshape(std::vector<int64_t> new_shape) const;
    Tensor cast(DType new_dtype) const;

private:
    std::vector<int64_t> shape_;
    DType dtype_ = DType::FLOAT32;
    TensorLayout layout_ = TensorLayout::NCHW;
    std::optional<QuantParams> quant_;
    void* data_ = nullptr;
    size_t size_bytes_ = 0;
    int ion_fd_ = -1; // ION/DMA-BUF 文件描述符（-1 表示普通内存）
};

// ============ 错误类型 ============
enum class ErrorCode : uint32_t {
    SUCCESS = 0,

    // 模型相关 (1000+)
    MODEL_NOT_FOUND      = 1001,
    MODEL_INVALID_FORMAT = 1002,
    MODEL_INCOMPATIBLE   = 1003,
    MODEL_CORRUPTED      = 1004,

    // 推理相关 (2000+)
    INFER_TIMEOUT        = 2001,
    INFER_OOM            = 2002,
    INFER_BACKEND_ERROR  = 2003,
    INFER_SHAPE_MISMATCH = 2004,

    // 硬件相关 (3000+)
    HARDWARE_NOT_AVAIL   = 3001,
    HARDWARE_INIT_FAIL   = 3002,
    HARDWARE_OVERHEAT    = 3003,

    // 内存相关 (4000+)
    MEMORY_OOM           = 4001,
    MEMORY_ALLOC_FAIL    = 4002,
};

template<typename T>
class [[nodiscard]] Result {
public:
    static Result<T> ok(T value);
    static Result<T> err(ErrorCode code, std::string message);

    bool is_ok() const;
    bool is_err() const;

    T& unwrap();                     // 断言 is_ok()，否则终止
    const T& unwrap() const;
    T unwrap_or(T fallback) const;  // 出错时返回 fallback

    ErrorCode error_code() const;
    const std::string& error_message() const;
};

} // namespace qoocore
```

### 2.2 推理流程（C++ 完整示例）

```cpp
// === 示例：目标检测推理 ===
#include <qoocore/engine.h>
#include <qoocore/tensor.h>
#include <iostream>

int main() {
    // 1. 初始化引擎
    qoocore::EngineConfig config;
    config.enable_profiling = true;
    config.allowed_backends = {
        qoocore::BackendType::NPU,
        qoocore::BackendType::GPU // NPU 不可用时的 fallback
    };
    config.max_memory_bytes = 2ULL * 1024 * 1024 * 1024; // 2GB

    auto& engine = qoocore::InferenceEngine::instance();
    if (auto result = engine.init(config); result.is_err()) {
        std::cerr << "引擎初始化失败: "
                  << result.error_message() << std::endl;
        return 1;
    }

    // 2. 加载模型
    qoocore::ModelConfig model_config;
    model_config.preferred_backend = qoocore::BackendType::NPU;
    model_config.enable_zero_copy = true;

    auto model_handle = engine.load_model(
        "yolov11n.qoomodel", model_config);
    if (model_handle.is_err()) {
        std::cerr << "模型加载失败: "
                  << model_handle.error_message() << std::endl;
        return 1;
    }

    // 3. 准备输入（从相机或文件）
    auto input_tensor = qoocore::Tensor::from_host(
        {1, 3, 640, 640}, qoocore::DType::FLOAT32,
        image_data, image_size);

    // 4. 同步推理
    auto output = engine.infer(model_handle.unwrap(), input_tensor);
    if (output.is_err()) {
        std::cerr << "推理失败: "
                  << output.error_message() << std::endl;
        return 1;
    }

    // 5. 解析输出
    const auto* detections = static_cast<const float*>(
        output.unwrap().data());
    // ... 后处理 NMS ...

    // 6. 异步推理（高吞吐量场景）
    std::vector<std::future<qoocore::Result<qoocore::Tensor>>> futures;
    for (const auto& img : batch_images) {
        futures.push_back(engine.infer_async(
            model_handle.unwrap(), img));
    }
    for (auto& f : futures) {
        auto result = f.get();
        // 处理结果...
    }

    // 7. 清理
    engine.unload_model(model_handle.unwrap());
    engine.shutdown();
    return 0;
}
```

### 2.3 编译 API（C++）

```cpp
// === 示例：模型编译为 .qoomodel ===
#include <qoocore/compiler.h>

int main() {
    qoocore::Compiler compiler;

    // 1. 导入 ONNX 模型
    auto import_result = compiler.import_onnx("yolov11n.onnx");
    if (import_result.is_err()) {
        std::cerr << "导入失败: "
                  << import_result.error_message() << std::endl;
        return 1;
    }
    auto model = std::move(import_result.unwrap());

    // 2. 图优化
    qoocore::OptimizeConfig opt_config;
    opt_config.enable_fusion     = true;  // Conv+BN+ReLU 融合
    opt_config.enable_const_fold = true;  // 常量折叠
    opt_config.enable_dce        = true;  // 死代码消除
    model = compiler.optimize(std::move(model), opt_config);

    // 3. 量化（PTQ，需要校准数据集）
    qoocore::QuantConfig quant_config;
    quant_config.dtype          = qoocore::DType::INT8;
    quant_config.calib_dir      = "./calib_dataset/";
    quant_config.calib_samples  = 100;
    quant_config.per_channel    = true;
    model = compiler.quantize(std::move(model), quant_config);

    // 4. 编译为目标后端
    qoocore::CompileConfig compile_config;
    compile_config.target_backend  = qoocore::BackendType::NPU;
    compile_config.optimization_level = 3; // 0~3
    compile_config.output_path   = "yolov11n.qoomodel";

    auto compile_result = compiler.compile(std::move(model),
                                          compile_config);
    if (compile_result.is_err()) {
        std::cerr << "编译失败: "
                  << compile_result.error_message() << std::endl;
        return 1;
    }

    std::cout << "编译成功: " << compile_config.output_path
              << std::endl;
    return 0;
}
```

## 4. CLI 工具设计

### 4.1 命令总览

```bash
qoocore --help

Usage: qoocore <COMMAND> [OPTIONS]

Commands:
  compile     编译模型为 .qoomodel 包
  infer       执行模型推理（支持单张/批量）
  profile     性能剖析（逐层延迟 + 火焰图）
  list        列出支持的芯片/后端/算子
  info        显示模型信息（输入输出/量化/大小）
  config      配置 qoocore（交互式或文件）
  version     显示版本和编译信息
  validate    验证 .qoomodel 文件完整性

Options:
  -h, --help     显示帮助
  -v, --verbose  详细输出
  -q, --quiet    静默模式
```

### 4.2 `compile` 命令

```bash
# 基本用法：编译 ONNX 为 .qoomodel（自动选择 NPU 后端）
qoocore compile --input yolov11n.onnx --output yolov11n.qoomodel

# 指定目标后端和量化精度
qoocore compile \
  --input yolov11n.onnx \
  --output yolov11n.qoomodel \
  --target npu \
  --quant int8 \
  --calib-dir ./calib_dataset/

# 指定优化级别和输出 IR
qoocore compile \
  --input model.onnx \
  --output model.qoomodel \
  --target gpu \
  --opt-level 3 \
  --dump-ir ir.mlir   # 导出设备 IR（调试用）

# 批量编译（通配符）
qoocore compile --input models/*.onnx --output models/ --target npu
```

**参数说明：**

| 参数 | 说明 | 默认值 |
|------|------|--------|
| `--input` | 输入模型路径（ONNX/PyTorch/TFLite） | 必需 |
| `--output` | 输出 `.qoomodel` 路径 | 必需 |
| `--target` | 目标后端：`npu`/`gpu`/`dsp`/`cpu`/`auto` | `auto` |
| `--quant` | 量化精度：`fp32`/`fp16`/`int8`/`int4` | `fp16` |
| `--calib-dir` | PTQ 校准数据集目录 | 无（fp32/fp16 不需要） |
| `--opt-level` | 优化级别：0~3 | 2 |
| `--dump-ir` | 导出设备 IR 到文件（调试） | 无 |

### 4.3 `infer` 命令

```bash
# 基本推理
qoocore infer --model yolov11n.qoomodel --input image.jpg

# 指定后端（覆盖模型默认配置）
qoocore infer \
  --model yolov11n.qoomodel \
  --input image.jpg \
  --backend gpu

# 性能剖析模式（输出逐层延迟）
qoocore infer \
  --model yolov11n.qoomodel \
  --input image.jpg \
  --profile \
  --profile-output profile.json

# 批量推理
qoocore infer \
  --model yolov11n.qoomodel \
  --input images/*.jpg \
  --batch-size 4 \
  --output results/

# 零拷贝模式（需要支持 ION 的平台）
qoocore infer \
  --model yolov11n.qoomodel \
  --input camera://0  # 从相机 ISP 直连
  --zero-copy
```

### 5.1 Proto 定义

```protobuf
// === qoocore.proto ===
syntax = "proto3";

package qoocore;

service InferenceService {
    // 模型管理
    rpc LoadModel    (LoadModelRequest)    returns (LoadModelResponse);
    rpc UnloadModel  (UnloadModelRequest)  returns (UnloadModelResponse);
    rpc GetModelInfo (GetModelInfoRequest) returns (ModelInfoResponse);
    rpc ListModels   (Empty)              returns (ListModelsResponse);

    // 推理
    rpc Infer        (InferRequest)        returns (InferResponse);
    rpc InferStream  (stream InferRequest) returns (stream InferResponse);

    // 性能
    rpc GetProfile   (GetProfileRequest)   returns (ProfileResponse);
}

message LoadModelRequest {
    string model_path = 1;
    string preferred_backend = 2; // "npu" / "gpu" / "auto"
    bool enable_profiling = 3;
}

message InferRequest {
    uint64 model_handle = 1;
    repeated TensorProto inputs = 2;
}

message TensorProto {
    repeated int64 shape = 1;
    string dtype = 2; // "float32" / "int8" / ...
    bytes data = 3;   // 张量数据（小 tensor）或
    string ion_fd_ref = 4; // ION 文件描述符引用（大 tensor，零拷贝）
}
```

## 6. 配置系统设计

### 6.1 配置文件格式（YAML）

默认搜索路径（按优先级）：
1. 当前目录 `./qoocore.yaml`
2. 用户目录 `~/.qoocore/qoocore.yaml`
3. 系统目录 `/etc/qoocore/qoocore.yaml`

```yaml
# qoocore.yaml
engine:
  log_level: info          # debug | info | warn | error
  enable_profiling: false
  max_models: 10           # 最多同时加载模型数
  profiling_output_dir: ./profiling

backend:
  npu:
    enabled: true
    preferred: true
    devices:
      - name: "Qualcomm NPU"
        hal: "qnn"         # qnn | bpu | rknn
        config:
          power_mode: 0    # 0=balanced, 1=high_perf, 2=power_saver
          max_memory_mb: 512
    fallback: "gpu"        # NPU 不可用时的 fallback

  gpu:
    enabled: true
    preferred: false
    backend: "cuda"        # cuda | opencl | vulkan
    device_id: 0
    max_memory_mb: 1024

  cpu:
    enabled: true
    threads: 0             # 0 = auto-detect (ARM big.LITTLE 感知)
    use_neon: true

memory:
  enable_zero_copy: true   # ION/DMA-BUF
  enable_model_pool: true  # 多模型共享权重
  enable_weight_compress: false
  max_memory_mb: 2048

compiler:
  default_target: "auto"   # 自动检测最佳后端
  optimization_level: 2
  enable_quantization: true
  calibration_samples: 100

profiler:
  enable: false
  detail_level: "layer"    # "model" | "layer" | "op"
  output_format: "json"    # "json" | "html" | "text"
```

## 7. 错误处理设计

### 7.1 错误码完整定义

```cpp
enum class ErrorCode : uint32_t {
    SUCCESS = 0,

    // === 模型相关 (1000-1999) ===
    MODEL_NOT_FOUND      = 1001, // 文件不存在
    MODEL_INVALID_FORMAT = 1002, // 不是有效的 .qoomodel
    MODEL_INCOMPATIBLE   = 1003, // 后端不兼容（如 NPU 模型在 GPU 上加载）
    MODEL_CORRUPTED      = 1004, // 校验和失败
    MODEL_VERSION_MISMATCH = 1005, // .qoomodel 版本过旧

    // === 推理相关 (2000-2999) ===
    INFER_TIMEOUT        = 2001, // 超过 timeout_ms
    INFER_OOM            = 2002, // 内存不足
    INFER_BACKEND_ERROR  = 2003, // 后端特定错误（附带后端错误信息）
    INFER_SHAPE_MISMATCH = 2004, // 输入形状与模型期望不符
    INFER_DTYPE_MISMATCH = 2005, // 输入 dtype 与模型期望不符
    INFER_NAN_DETECTED   = 2006, // 推理输出包含 NaN

    // === 硬件相关 (3000-3999) ===
    HARDWARE_NOT_AVAIL   = 3001, // 后端硬件不存在
    HARDWARE_INIT_FAIL   = 3002, // 硬件初始化失败（驱动问题）
    HARDWARE_OVERHEAT    = 3003, // 温度过高，触发热保护
    HARDWARE_OUT_OF_MEM  = 3004, // NPU/DSP 内部内存不足

    // === 内存相关 (4000-4999) ===
    MEMORY_OOM           = 4001, // 主机内存不足
    MEMORY_ALLOC_FAIL    = 4002, // 内存分配系统调用失败
    MEMORY_ION_FAIL      = 4003, // ION 分配失败

    // === 编译器相关 (5000-5999) ===
    COMPILER_IMPORT_FAIL = 5001, // 模型导入失败（格式不支持）
    COMPILER_IR_FAIL     = 5002, // IR 构建失败
    COMPILER_QUANT_FAIL  = 5003, // 量化失败（校准数据不足）
    COMPILER_CODEGEN_FAIL= 5004, // 代码生成失败（后端不支持某算子）
};
```

### 7.2 错误恢复建议

| 错误码 | 自动恢复策略 | 用户建议 |
|--------|-------------|---------|
| `HARDWARE_NOT_AVAIL` | 自动 fallback 到下一个后端 | 检查硬件驱动 |
| `INFER_OOM` | 释放不常用模型后再试 | 减小 `max_models` |
| `MODEL_INCOMPATIBLE` | 提示重新编译模型 | 运行 `qoocore compile --target auto` |
| `COMPILER_CODEGEN_FAIL` | 回退不支持的算子到 CPU | 检查算子支持列表 |

## 8. 开发者工作流程

```
┌──────────────────────────────────────────────────────────────────┐
│                   模型开发阶段                                     │
│                                                                    │
│  1. 训练模型 (PyTorch/TensorFlow)                                 │
│     ↓                                                              │
│  2. 导出为 ONNX                                                   │
│     torch.onnx.export(model, ..., "model.onnx")                  │
│     ↓                                                              │
│  3. 编译为 .qoomodel                                              │
│     qoocore compile --input model.onnx \                         │
│       --output model.qoomodel --target npu --quant int8          │
│     ↓                                                              │
│  4. 验证编译结果                                                   │
│     qoocore info --model model.qoomodel                          │
│     qoocore infer --model model.qoomodel --input test.jpg        │
│                                                                    │
└────────────────────────────┬─────────────────────────────────────┘
                             │
                             ▼
┌──────────────────────────────────────────────────────────────────┐
│                   部署阶段                                         │
│                                                                    │
│  1. 将 .qoomodel 部署到端侧设备                                    │
│     scp model.qoomodel robot:/opt/qoobot/models/                 │
│     ↓                                                              │
│  2. 配置 qoocore.yaml                                             │
│     (设置后端优先级、内存限制等)                                    │
│     ↓                                                              │
│  3. 验证部署                                                       │
│     ssh robot "qoocore infer --model /opt/qoobot/models/model.qoomodel \│
│       --input /dev/video0 --profile"                             │
│                                                                    │
└────────────────────────────┬─────────────────────────────────────┘
                             │
                             ▼
┌──────────────────────────────────────────────────────────────────┐
│                   运行阶段                                         │
│                                                                    │
│  1. qoobrain 调用 qoocore C++ API                                 │
│     auto result = engine.infer(model, sensor_data);              │
│     ↓                                                              │
│  2. 定期性能剖析（开发阶段）                                       │
│     qoocore profile --model model.qoomodel --backend all \       │
│       --output flamegraph.html                                   │
│     ↓                                                              │
│  3. 模型更新（OTA）                                               │
│     qoocore ota --model model.qoomodel --url https://...         │
│                                                                    │
└──────────────────────────────────────────────────────────────────┘
```
