## 4. 自动化测试

### 4.1 WebRTC 模拟测试

```python
# 使用 aiortc 进行 WebRTC 自动化测试
async def test_teleop_session():
    # 创建模拟机器人端
    robot = await create_robot_peer()
    # 创建模拟操控端
    client = await create_client_peer()
    
    # 建立 WebRTC 连接
    await establish_connection(client, robot)
    
    # 验证视频流
    frame = await client.receive_video_frame(timeout=5)
    assert frame.width == 1280
    assert frame.height == 720
    
    # 发送控制指令
    await client.send_control({"joint": "head_pan", "position": 0.5})
    
    # 验证机器人端收到
    cmd = await robot.receive_command(timeout=1)
    assert cmd["joint"] == "head_pan"
    assert abs(cmd["position"] - 0.5) < 0.001
```

### 4.2 安全看门狗测试

```cpp
TEST(SafetyWatch, HeartbeatTimeoutTriggersStop) {
    SafetyWatch watch;
    watch.setHeartbeatTimeout(500ms);
    
    // 模拟 600ms 无心跳
    std::this_thread::sleep_for(600ms);
    
    ASSERT_TRUE(watch.isEmergencyStopped());
    ASSERT_EQ(watch.getState(), RobotState::EMERGENCY_STOP);
}

TEST(SafetyWatch, JointLimitViolation) {
    SafetyWatch watch;
    JointLimits limits;
    limits["arm_elbow"] = {-1.5, 1.5, 2.0, 50.0}; // min, max, vel, torque
    
    watch.loadJointLimits(limits);
    
    JointCmd cmd{"arm_elbow", 2.0, 0.5, 10.0}; // position exceeds max
    ASSERT_FALSE(watch.validateJointCommand(cmd));
}
```
