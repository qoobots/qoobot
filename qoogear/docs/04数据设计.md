## 2. 数据库表设计

### 2.1 开发者表

```sql
CREATE TABLE developers (
    id              BIGSERIAL PRIMARY KEY,
    user_id         BIGINT NOT NULL UNIQUE,          -- qooauth 用户ID
    company_name    VARCHAR(200) NOT NULL,
    contact_name    VARCHAR(100) NOT NULL,
    contact_email   VARCHAR(255) NOT NULL,
    contact_phone   VARCHAR(50),
    website         VARCHAR(500),
    country         VARCHAR(100),
    business_license VARCHAR(500),                   -- 营业执照 MinIO URL
    verified_at     TIMESTAMPTZ,                     -- 企业认证通过时间
    verification_status VARCHAR(20) DEFAULT 'pending', -- pending/verified/rejected
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_developers_user_id ON developers(user_id);
CREATE INDEX idx_developers_verified ON developers(verification_status);
```

### 2.2 认证申请表

```sql
CREATE TABLE cert_applications (
    id              BIGSERIAL PRIMARY KEY,
    developer_id    BIGINT NOT NULL REFERENCES developers(id),
    product_name    VARCHAR(200) NOT NULL,
    product_category VARCHAR(50) NOT NULL,            -- gripper/sensor/wearable/power/mobility/tool
    product_model   VARCHAR(100) NOT NULL,
    product_description TEXT,
    cert_level      VARCHAR(20) NOT NULL,              -- basic/premium/pro
    standard_ids    BIGINT[] NOT NULL,                 -- 适用的接口标准
    mechanical_spec_url  VARCHAR(500),                 -- MinIO 文件URL
    electrical_spec_url  VARCHAR(500),
    communication_protocol_url VARCHAR(500),
    firmware_source_url   VARCHAR(500),
    test_samples_count    INT DEFAULT 2,
    status          VARCHAR(30) NOT NULL DEFAULT 'draft',
    -- draft/submitted/compliance_check/reviewing/assigned/testing/
    -- test_completed/security_review/approved/rejected/revoked/expired
    submitted_at    TIMESTAMPTZ,
    reviewed_by     BIGINT,                           -- 审核员 user_id
    review_comment  TEXT,
    reviewed_at     TIMESTAMPTZ,
    assigned_lab_id BIGINT REFERENCES laboratories(id),
    assigned_at     TIMESTAMPTZ,
    approved_at     TIMESTAMPTZ,
    rejection_reason TEXT,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_cert_app_developer ON cert_applications(developer_id);
CREATE INDEX idx_cert_app_status ON cert_applications(status);
CREATE INDEX idx_cert_app_category ON cert_applications(product_category);
CREATE INDEX idx_cert_app_lab ON cert_applications(assigned_lab_id);
```

### 2.3 证书表

```sql
CREATE TABLE certificates (
    id              BIGSERIAL PRIMARY KEY,
    application_id  BIGINT NOT NULL UNIQUE REFERENCES cert_applications(id),
    cert_number     VARCHAR(50) NOT NULL UNIQUE,       -- MFQ-2026-XXXXXX
    cert_level      VARCHAR(20) NOT NULL,
    developer_id    BIGINT NOT NULL REFERENCES developers(id),
    product_name    VARCHAR(200) NOT NULL,
    product_model   VARCHAR(100) NOT NULL,
    product_category VARCHAR(50) NOT NULL,
    issued_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    expires_at      TIMESTAMPTZ NOT NULL,              -- 颁发后 2 年
    revoked_at      TIMESTAMPTZ,
    revoke_reason   TEXT,
    public_key      TEXT,                              -- 证书公钥
    cert_doc_url    VARCHAR(500),                      -- 证书PDF MinIO URL
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_certificates_number ON certificates(cert_number);
CREATE INDEX idx_certificates_developer ON certificates(developer_id);
CREATE INDEX idx_certificates_expires ON certificates(expires_at);
CREATE INDEX idx_certificates_active ON certificates(revoked_at)
    WHERE revoked_at IS NULL;
```

### 2.4 授权芯片表

```sql
CREATE TABLE auth_chips (
    id              BIGSERIAL PRIMARY KEY,
    certificate_id  BIGINT NOT NULL REFERENCES certificates(id),
    chip_id         VARCHAR(100) NOT NULL UNIQUE,      -- 芯片唯一ID
    chip_type       VARCHAR(50) NOT NULL,              -- DS28E38/NXP A1006/Infineon OPTIGA
    batch_number    VARCHAR(100) NOT NULL,
    burned_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    status          VARCHAR(20) NOT NULL DEFAULT 'active', -- active/revoked/lost
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_auth_chips_cert ON auth_chips(certificate_id);
CREATE INDEX idx_auth_chips_batch ON auth_chips(batch_number);
```

### 2.5 测试报告表

```sql
CREATE TABLE test_reports (
    id              BIGSERIAL PRIMARY KEY,
    application_id  BIGINT NOT NULL REFERENCES cert_applications(id),
    laboratory_id   BIGINT NOT NULL REFERENCES laboratories(id),
    test_engineer   VARCHAR(100),
    overall_result  VARCHAR(20) NOT NULL,              -- pass/fail/conditional_pass
    summary         TEXT,
    test_data_json  JSONB,                             -- 详细测试数据
    attachments     VARCHAR(500)[],                     -- MinIO 文件URL数组
    submitted_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_test_reports_app ON test_reports(application_id);
CREATE INDEX idx_test_reports_lab ON test_reports(laboratory_id);
```

### 2.6 测试检查项表

```sql
CREATE TABLE test_checklist (
    id              BIGSERIAL PRIMARY KEY,
    standard_id     BIGINT NOT NULL REFERENCES standard_specs(id),
    test_item       VARCHAR(200) NOT NULL,
    test_method     TEXT,
    criteria        TEXT NOT NULL,
    required        BOOLEAN NOT NULL DEFAULT true,
    sort_order      INT DEFAULT 0,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_test_checklist_standard ON test_checklist(standard_id);
```

### 2.7 接口规范表

```sql
CREATE TABLE standard_specs (
    id              BIGSERIAL PRIMARY KEY,
    category_id     BIGINT NOT NULL REFERENCES standard_categories(id),
    title           VARCHAR(300) NOT NULL,
    spec_number     VARCHAR(50) NOT NULL,              -- MFQ-SPEC-XXXX
    version         VARCHAR(20) NOT NULL,              -- SemVer
    status          VARCHAR(20) NOT NULL DEFAULT 'draft',
    -- draft/published/deprecated/retired
    description     TEXT,
    spec_doc_url    VARCHAR(500) NOT NULL,              -- 规范文档 MinIO URL
    changelog       TEXT,
    published_at    TIMESTAMPTZ,
    deprecated_at   TIMESTAMPTZ,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE(spec_number, version)
);

CREATE INDEX idx_standard_specs_category ON standard_specs(category_id);
CREATE INDEX idx_standard_specs_status ON standard_specs(status);
```

### 2.8 标准类别表

```sql
CREATE TABLE standard_categories (
    id              BIGSERIAL PRIMARY KEY,
    name            VARCHAR(100) NOT NULL,
    slug            VARCHAR(100) NOT NULL UNIQUE,
    parent_id       BIGINT REFERENCES standard_categories(id),
    description     TEXT,
    icon            VARCHAR(50),
    sort_order      INT DEFAULT 0,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_std_categories_parent ON standard_categories(parent_id);
```

### 2.9 兼容性矩阵表

```sql
CREATE TABLE compatibility_matrix (
    id              BIGSERIAL PRIMARY KEY,
    spec_id_a       BIGINT NOT NULL REFERENCES standard_specs(id),
    spec_version_a  VARCHAR(20) NOT NULL,
    spec_id_b       BIGINT NOT NULL REFERENCES standard_specs(id),
    spec_version_b  VARCHAR(20) NOT NULL,
    compatibility   VARCHAR(20) NOT NULL,              -- compatible/incompatible/conditional
    condition_desc  TEXT,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE(spec_id_a, spec_version_a, spec_id_b, spec_version_b)
);
```

### 2.10 实验室表

```sql
CREATE TABLE laboratories (
    id              BIGSERIAL PRIMARY KEY,
    name            VARCHAR(200) NOT NULL,
    lab_code         VARCHAR(50) NOT NULL UNIQUE,       -- LAB-XXX
    country         VARCHAR(100) NOT NULL,
    city            VARCHAR(100),
    address         TEXT NOT NULL,
    contact_name    VARCHAR(100) NOT NULL,
    contact_email   VARCHAR(255) NOT NULL,
    contact_phone   VARCHAR(50),
    accreditation   VARCHAR(200),                      -- ISO 17025 等
    scope           VARCHAR(100)[],                     -- 测试范围：mechanical/electrical/communication/safety
    status          VARCHAR(20) NOT NULL DEFAULT 'active', -- active/suspended/revoked
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_laboratories_scope ON laboratories USING GIN(scope);
CREATE INDEX idx_laboratories_status ON laboratories(status);
```

### 2.11 实验室设备表

```sql
CREATE TABLE lab_equipment (

    calibrated_at   TIMESTAMPTZ,
    next_calibration_due TIMESTAMPTZ,

    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()

CREATE INDEX idx_lab_equipment_lab ON lab_equipment(laboratory_id);
CREATE INDEX idx_lab_equipment_cal ON lab_equipment(next_calibration_due);
```

### 2.12 安全审计表

```sql
CREATE TABLE security_audits (
    id              BIGSERIAL PRIMARY KEY,
    application_id  BIGINT NOT NULL REFERENCES cert_applications(id),
    risk_level      VARCHAR(20) NOT NULL,              -- low/medium/high/critical
    fmea_json       JSONB,                             -- FMEA 分析结构化数据
    auditor_id      BIGINT NOT NULL,                   -- 安全审核员 user_id
    findings        TEXT,
    recommendation  TEXT,
    status          VARCHAR(20) NOT NULL DEFAULT 'pending', -- pending/approved/rejected
    completed_at    TIMESTAMPTZ,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_security_audits_app ON security_audits(application_id);
CREATE INDEX idx_security_audits_risk ON security_audits(risk_level);
```

### 2.14 SDK 发布表

```sql
CREATE TABLE sdk_releases (
    id              BIGSERIAL PRIMARY KEY,
    platform        VARCHAR(20) NOT NULL,              -- python/cpp
    version         VARCHAR(20) NOT NULL,
    download_url    VARCHAR(500) NOT NULL,
    file_size       BIGINT,
    checksum_sha256 VARCHAR(64),
    release_notes   TEXT,
    is_latest       BOOLEAN DEFAULT false,
    released_at     TIMESTAMPTZ NOT NULL DEFAULT NOW(),

);

CREATE INDEX idx_sdk_releases_platform ON sdk_releases(platform);
CREATE INDEX idx_sdk_releases_latest ON sdk_releases(platform) WHERE is_latest = true;
```

### 2.16 审计日志表

```sql
CREATE TABLE audit_logs (
```

## 3. Redis 缓存设计

| Key 模式 | 类型 | TTL | 说明 |
|----------|------|-----|------|
| `qoogear:cert:active:{certNumber}` | Hash | 1h | 活跃证书缓存 |
| `qoogear:cert:verify:{certNumber}` | String | 5min | 证书验证结果 |
| `qoogear:standard:spec:{id}` | Hash | 1h | 接口规范缓存 |
| `qoogear:standard:category:tree` | String(JSON) | 30min | 标准类别树 |
| `qoogear:developer:{id}:dashboard` | Hash | 5min | 开发者仪表板数据 |
| `qoogear:lab:{id}:workload` | Hash | 5min | 实验室工作量统计 |
| `qoogear:hot:references` | ZSet | 1h | 热门参考设计 |
| `qoogear:session:{token}` | Hash | 30min | 用户会话 |

## 5. RocketMQ 消息主题

| Topic | 说明 | 消费者 |
|-------|------|--------|
| `qoogear-cert-application-submitted` | 认证申请已提交 | 审核员通知 |
| `qoogear-cert-application-approved` | 认证申请已批准 | 证书生成/开发者通知 |
| `qoogear-cert-application-rejected` | 认证申请已驳回 | 开发者通知 |
| `qoogear-cert-issued` | 证书已颁发 | qoostore 同步/芯片烧录 |
| `qoogear-cert-revoked` | 证书已吊销 | qoostore 下架/芯片吊销 |
| `qoogear-cert-expiring` | 证书即将过期 | 开发者提醒 |
| `qoogear-standard-published` | 新标准版本发布 | 开发者通知 |
| `qoogear-peripheral-identified` | 配件被识别 | 驱动匹配/能力注册 |
