## 2. 数据库设计 (PostgreSQL)

### 2.1 产品与 BOM

```sql
-- 产品型号
CREATE TABLE product (
    id              BIGSERIAL PRIMARY KEY,
    model_code      VARCHAR(64)  NOT NULL UNIQUE,    -- 产品型号编码 e.g. QooBot-Pro
    model_name      VARCHAR(128) NOT NULL,            -- 产品名称
    category        VARCHAR(32)  NOT NULL,            -- 分类: HUMANOID/MOBILE/ARM
    status          VARCHAR(16)  NOT NULL DEFAULT 'DRAFT',
                                                     -- DRAFT/RELEASED/OBSOLETE
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- BOM 版本
CREATE TABLE bom (
    id              BIGSERIAL PRIMARY KEY,
    product_id      BIGINT       NOT NULL REFERENCES product(id),
    version         VARCHAR(16)  NOT NULL,            -- BOM 版本号 e.g. v2.1
    bom_type        VARCHAR(8)   NOT NULL,            -- EBOM/MBOM
    status          VARCHAR(16)  NOT NULL DEFAULT 'DRAFT',
                                                     -- DRAFT/UNDER_REVIEW/RELEASED/FROZEN
    total_items     INT          NOT NULL DEFAULT 0,
    estimated_cost  NUMERIC(12,2),                    -- 预估成本 (CNY)
    cost_currency   VARCHAR(3)   DEFAULT 'CNY',
    released_at     TIMESTAMPTZ,
    created_by      VARCHAR(64)  NOT NULL,
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    UNIQUE(product_id, version, bom_type)
);

-- BOM 物料项 (树形结构通过 parent_item_id 实现)
CREATE TABLE bom_item (
    id              BIGSERIAL PRIMARY KEY,
    bom_id          BIGINT       NOT NULL REFERENCES bom(id),
    parent_item_id  BIGINT       REFERENCES bom_item(id),  -- NULL = 根节点
    item_code       VARCHAR(64)  NOT NULL,            -- 物料编码
    item_name       VARCHAR(128) NOT NULL,            -- 物料名称
    level           INT          NOT NULL DEFAULT 0,  -- 层级: 0=整机 1=模组 2=组件 3=器件
    quantity        NUMERIC(10,3) NOT NULL DEFAULT 1, -- 用量
    unit            VARCHAR(16)  NOT NULL DEFAULT 'PCS',
    material_id     BIGINT       REFERENCES material(id),
    is_critical     BOOLEAN      NOT NULL DEFAULT FALSE, -- 关键器件
    sort_order      INT          NOT NULL DEFAULT 0,
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_bom_item_bom ON bom_item(bom_id);
CREATE INDEX idx_bom_item_parent ON bom_item(parent_item_id);
```

### 2.2 物料与供应商

```sql
-- 物料主数据
CREATE TABLE material (
    id              BIGSERIAL PRIMARY KEY,
    material_code   VARCHAR(64)  NOT NULL UNIQUE,     -- 统一物料编码
    material_name   VARCHAR(128) NOT NULL,
    category        VARCHAR(32)  NOT NULL,            -- SOC/SENSOR/MOTOR/BATTERY/PCB/CABLE/...
    specification   TEXT,                              -- 规格描述
    manufacturer    VARCHAR(128),                      -- 原厂
    manufacturer_pn VARCHAR(64),                       -- 原厂型号
    lifecycle       VARCHAR(16)  NOT NULL DEFAULT 'ACTIVE',
                                                     -- ACTIVE/NRND/EOL/OBSOLETE
    lead_time_days  INT,                               -- 标准交期(天)
    moq             INT,                               -- 最小起订量
    rohs_compliant  BOOLEAN      DEFAULT FALSE,
    reach_compliant BOOLEAN      DEFAULT FALSE,
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- 供应商
CREATE TABLE supplier (
    id              BIGSERIAL PRIMARY KEY,
    supplier_code   VARCHAR(64)  NOT NULL UNIQUE,     -- SUP001
    supplier_name   VARCHAR(128) NOT NULL,
    category        VARCHAR(32)  NOT NULL,            -- SENSOR/MOTOR/BATTERY/PCB/...
    country         VARCHAR(64),
    rating          INT          NOT NULL DEFAULT 3,  -- 1-5 星
    status          VARCHAR(16)  NOT NULL DEFAULT 'QUALIFIED',
                                                     -- TRIAL/QUALIFIED/PROBATION/DISQUALIFIED
    contact_name    VARCHAR(64),
    contact_email   VARCHAR(128),
    contact_phone   VARCHAR(32),
    audit_date      DATE,
    audit_report_url VARCHAR(512),
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- 物料-供应商关联 (AVL 合格供应商清单)
CREATE TABLE material_supplier (
    id              BIGSERIAL PRIMARY KEY,
    material_id     BIGINT       NOT NULL REFERENCES material(id),
    supplier_id     BIGINT       NOT NULL REFERENCES supplier(id),
    supplier_pn     VARCHAR(64),                       -- 供应商型号
    unit_price      NUMERIC(10,2),                     -- 单价
    currency        VARCHAR(3)   DEFAULT 'CNY',
    is_preferred    BOOLEAN      DEFAULT FALSE,        -- 首选供应商
    qualification_status VARCHAR(16) DEFAULT 'QUALIFIED',
    qualified_at    DATE,
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    UNIQUE(material_id, supplier_id)
);

-- 替代料关系
CREATE TABLE material_alternative (
    id              BIGSERIAL PRIMARY KEY,
    material_id     BIGINT       NOT NULL REFERENCES material(id),
    alternative_id  BIGINT       NOT NULL REFERENCES material(id),
    compatibility   VARCHAR(16)  NOT NULL,             -- DROP_IN/FORM_FIT/MINOR_MOD/MAJOR_MOD
    verified        BOOLEAN      DEFAULT FALSE,
    verified_at     TIMESTAMPTZ,
    notes           TEXT,
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    UNIQUE(material_id, alternative_id),
    CHECK(material_id <> alternative_id)
);

-- 成本记录
CREATE TABLE cost_record (
    id              BIGSERIAL PRIMARY KEY,
    material_id     BIGINT       NOT NULL REFERENCES material(id),
    supplier_id     BIGINT       NOT NULL REFERENCES supplier(id),
    unit_price      NUMERIC(10,2) NOT NULL,
    currency        VARCHAR(3)   DEFAULT 'CNY',
    effective_from  DATE         NOT NULL,
    effective_to    DATE,
    change_reason   VARCHAR(64),
    recorded_at     TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_cost_material ON cost_record(material_id);
```

### 2.3 产线与制造

```sql
-- 产线
CREATE TABLE production_line (
    id              BIGSERIAL PRIMARY KEY,
    line_code       VARCHAR(64)  NOT NULL UNIQUE,     -- SZ-Line-01
    line_name       VARCHAR(128) NOT NULL,
    location        VARCHAR(128),
    product_model   VARCHAR(64)  NOT NULL,
    status          VARCHAR(16)  NOT NULL DEFAULT 'ACTIVE',
    takt_time_min   INT,                               -- 节拍时间(分钟)
    daily_capacity  INT,                               -- 日产能
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- 工位
CREATE TABLE station (
    id              BIGSERIAL PRIMARY KEY,
    line_id         BIGINT       NOT NULL REFERENCES production_line(id),
    station_code    VARCHAR(64)  NOT NULL,             -- ST01
    station_name    VARCHAR(128) NOT NULL,             -- 底盘装配
    sequence        INT          NOT NULL,             -- 工序顺序
    cycle_time_min  INT          NOT NULL,             -- 标准工时(分钟)
    tools_required  TEXT,                               -- 所需工具 JSON 数组
    poka_yoke_rules TEXT,                               -- 防错规则 JSON
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    UNIQUE(line_id, station_code)
);

-- SOP 步骤
CREATE TABLE sop_step (
    id              BIGSERIAL PRIMARY KEY,
    station_id      BIGINT       NOT NULL REFERENCES station(id),
    step_number     INT          NOT NULL,
    description     TEXT         NOT NULL,             -- 操作描述
    duration_min    INT          NOT NULL,             -- 工时
    tools           VARCHAR(256),                      -- 所需工具
    torque_spec     NUMERIC(8,2),                      -- 扭矩规格(Nm)
    inspection_point BOOLEAN     DEFAULT FALSE,        -- 是否需要检验
    image_url       VARCHAR(512),                      -- 操作图示
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    UNIQUE(station_id, step_number)
);

-- DFM/DFA 审查
CREATE TABLE dfm_check (
    id              BIGSERIAL PRIMARY KEY,
    product_id      BIGINT       NOT NULL REFERENCES product(id),
    checklist_item  VARCHAR(256) NOT NULL,             -- 审查项目
    category        VARCHAR(32)  NOT NULL,             -- DFM/DFA
    status          VARCHAR(16)  NOT NULL DEFAULT 'OPEN',
    severity        VARCHAR(8)   DEFAULT 'MEDIUM',     -- LOW/MEDIUM/HIGH/CRITICAL
    assignee        VARCHAR(64),
    resolution      TEXT,
    resolved_at     TIMESTAMPTZ,
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);
```

### 2.4 机器人实例与标定

```sql
-- 机器人实例 (每台出厂的物理机器人)
CREATE TABLE robot (
    id              BIGSERIAL PRIMARY KEY,
    serial_number   VARCHAR(64)  NOT NULL UNIQUE,     -- QB-PRO-2026-0628-0001
    product_id      BIGINT       NOT NULL REFERENCES product(id),
    hardware_version VARCHAR(16) NOT NULL,
    firmware_version VARCHAR(16),
    production_line_id BIGINT    REFERENCES production_line(id),
    status          VARCHAR(16)  NOT NULL DEFAULT 'MANUFACTURING',
                                                     -- MANUFACTURING/CALIBRATING/TESTING/FINISHED/SHIPPED
    manufactured_at DATE,
    shipped_at      TIMESTAMPTZ,
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- 标定会话
CREATE TABLE calibration_session (
    id              BIGSERIAL PRIMARY KEY,
    robot_id        BIGINT       NOT NULL REFERENCES robot(id),
    session_version VARCHAR(32)  NOT NULL,             -- cal-0628-003
    calib_type      VARCHAR(16)  NOT NULL,             -- CAMERA/IMU/LIDAR/KINEMATIC/FORCE/ALL
    operator_id     VARCHAR(64)  NOT NULL,
    status          VARCHAR(16)  NOT NULL DEFAULT 'IN_PROGRESS',
                                                     -- IN_PROGRESS/PASSED/FAILED
    started_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    completed_at    TIMESTAMPTZ,
    report_url      VARCHAR(512),
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- 标定结果
CREATE TABLE calibration_result (
    id              BIGSERIAL PRIMARY KEY,
    session_id      BIGINT       NOT NULL REFERENCES calibration_session(id),
    sensor_type     VARCHAR(32)  NOT NULL,             -- CAM_FRONT/CAM_LEFT/IMU/LIDAR/JOINT_01/...
    parameter_name  VARCHAR(64)  NOT NULL,             -- fx/fy/cx/cy/bias_x/...
    parameter_value NUMERIC(16,8) NOT NULL,
    unit            VARCHAR(16),
    accuracy_metric VARCHAR(64),                        -- RMS/STD_DEV/MAX_ERROR
    accuracy_value  NUMERIC(16,8),
    accuracy_unit   VARCHAR(16),
    spec_lower      NUMERIC(16,8),
    spec_upper      NUMERIC(16,8),
    passed          BOOLEAN      NOT NULL,
    raw_data_url    VARCHAR(512),                      -- MinIO 原始数据路径
    created_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_calib_session ON calibration_result(session_id);
```

### 2.5 质量检测

```sql
-- 检测记录
CREATE TABLE inspection_record (
    id              BIGSERIAL PRIMARY KEY,
    robot_id        BIGINT       REFERENCES robot(id),
```

## 4. Redis 缓存设计

| Key 模式 | 类型 | TTL | 说明 |
|----------|------|-----|------|
| `bom:{product}:{version}` | Hash | 1h | BOM 树缓存 |
| `material:{code}` | Hash | 24h | 物料主数据缓存 |
| `supplier:{code}` | Hash | 24h | 供应商信息缓存 |
| `calib:latest:{serial}` | String(JSON) | 1h | 最新标定参数 |

## 6. 数据安全

| 数据类别 | 敏感级别 | 访问控制 | 加密 |
|----------|:--------:|----------|------|
| BOM 结构 | 内部 | RBAC: 工艺工程师+ | - |
| 供应商报价/成本 | 机密 | RBAC: 供应链管理 | AES-256 存储加密 |
| 标定参数 | 内部 | RBAC: 标定工程师+ | - |
| 质检数据 | 内部 | RBAC: 质量经理+ | - |
| 数字护照 | 公开(用户可查看) | 基于 SN 查询 | Ed25519 签名防篡改 |
| 供应商合同 | 机密 | RBAC: 供应链管理 | AES-256 存储加密 |
| 操作员信息 | PII | RBAC: 管理员 | 脱敏显示 |
