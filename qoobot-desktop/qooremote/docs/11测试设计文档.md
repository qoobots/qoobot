## 6. CI/CD 流水线

```yaml
# .github/workflows/qooremote-test.yml
name: QooRemote Test

on:
  push:
    paths: ['qoobot-desktop/qooremote/**']
  pull_request:
    paths: ['qoobot-desktop/qooremote/**']

jobs:
  python-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with: { python-version: '3.11' }
      - run: pip install -e ".[dev]" && cd qoobot-desktop/qooremote && pytest tests/ --cov

  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with: { python-version: '3.11' }
      - run: pip install mypy ruff && cd qoobot-desktop/qooremote && ruff check . && mypy console/

  web-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: '20' }
      - run: cd qoobot-desktop/qooremote/web && npm ci && npm run test

  e2e:
    runs-on: ubuntu-latest
    needs: [python-test, web-test]
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with: { python-version: '3.11' }
      - run: pip install -e ".[dev]" && cd qoobot-desktop/qooremote && pytest tests/integration/
```
