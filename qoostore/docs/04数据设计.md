## 二、数据库表设计

### 2.1 开发者 (developers)

```sql
CREATE TABLE developers (
    id              BIGSERIAL PRIMARY KEY,
    user_id         UUID NOT NULL UNIQUE,          -- qooauth 用户 ID
    username        VARCHAR(64) NOT NULL UNIQUE,
    display_name    VARCHAR(128),
    email           VARCHAR(255) NOT NULL,
    company         VARCHAR(255),
    website         VARCHAR(512),
    verified        BOOLEAN DEFAULT FALSE,          -- 实名认证
    tax_id          VARCHAR(64),                    -- 税务 ID
    payout_method   VARCHAR(32),                    -- stripe/alipay/bank
    payout_account  VARCHAR(512),                   -- 收款账户(加密)
    status          VARCHAR(16) DEFAULT 'active',   -- active/suspended/banned
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_developers_user_id ON developers(user_id);
CREATE INDEX idx_developers_status ON developers(status);
```

### 2.2 技能 (skills)

```sql
CREATE TABLE skills (
    id              BIGSERIAL PRIMARY KEY,
    skill_id        VARCHAR(255) NOT NULL UNIQUE,   -- com.example.cleaning
    name            VARCHAR(128) NOT NULL,
    developer_id    BIGINT NOT NULL REFERENCES developers(id),
    category_id     BIGINT REFERENCES categories(id),
    tagline         VARCHAR(255),                   -- 一句话简介
    description     TEXT,
    icon_url        VARCHAR(1024),
    privacy_level   VARCHAR(16) DEFAULT 'public',   -- public/unlisted/private
    status          VARCHAR(16) DEFAULT 'draft',    -- draft/published/suspended/removed
    pricing_model   VARCHAR(32) DEFAULT 'free',     -- free/paid/subscription/usage
    price           DECIMAL(10,2) DEFAULT 0.00,
    currency        VARCHAR(3) DEFAULT 'USD',
    trial_days      INTEGER DEFAULT 0,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    published_at    TIMESTAMP
);

CREATE INDEX idx_skills_skill_id ON skills(skill_id);
CREATE INDEX idx_skills_developer_id ON skills(developer_id);
CREATE INDEX idx_skills_category_id ON skills(category_id);
CREATE INDEX idx_skills_status ON skills(status);
CREATE INDEX idx_skills_name_fts ON skills USING GIN(to_tsvector('english', name || ' ' || COALESCE(description, '')));
```

### 2.3 技能版本 (skill_versions)

```sql
CREATE TABLE skill_versions (
    id              BIGSERIAL PRIMARY KEY,
    skill_id        BIGINT NOT NULL REFERENCES skills(id),
    version         VARCHAR(32) NOT NULL,            -- 语义版本 1.2.3
    changelog       TEXT,
    package_url     VARCHAR(1024),                   -- MinIO 对象 URL
    package_size    BIGINT,                          -- 字节
    package_hash    VARCHAR(128),                    -- SHA-256
    manifest_json   JSONB,                           -- manifest.json 内容
    min_qos_version VARCHAR(16),                     -- 最低 QOS 版本
    privacy_label   JSONB,                           -- 隐私标签 JSON
    permissions     TEXT[],                          -- 权限声明数组
    status          VARCHAR(16) DEFAULT 'pending',   -- pending/reviewing/approved/rejected
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(skill_id, version)
);

CREATE INDEX idx_skill_versions_skill_id ON skill_versions(skill_id);
CREATE INDEX idx_skill_versions_status ON skill_versions(status);
```

### 2.4 审核提交 (submissions)

```sql
CREATE TABLE submissions (
    id              BIGSERIAL PRIMARY KEY,
    version_id      BIGINT NOT NULL REFERENCES skill_versions(id),
    type            VARCHAR(16) DEFAULT 'new',       -- new/update
    status          VARCHAR(16) DEFAULT 'pending',   -- pending/auto_reviewing/manual_reviewing/approved/rejected
    auto_scan_id    VARCHAR(64),                     -- 自动扫描任务 ID
    scan_result     JSONB,                           -- 扫描结果详情
    reviewer_id     UUID,                            -- 审核员 (qooauth user id)
    reject_reason   TEXT,
    reviewed_at     TIMESTAMP,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_submissions_version_id ON submissions(version_id);
CREATE INDEX idx_submissions_status ON submissions(status);
```

### 2.5 分类 (categories)

```sql
CREATE TABLE categories (
    id              BIGSERIAL PRIMARY KEY,
    name            VARCHAR(64) NOT NULL,
    slug            VARCHAR(64) NOT NULL UNIQUE,
    description     TEXT,
    icon            VARCHAR(255),
    parent_id       BIGINT REFERENCES categories(id),
    sort_order      INTEGER DEFAULT 0,
    status          VARCHAR(16) DEFAULT 'active',
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_categories_slug ON categories(slug);
CREATE INDEX idx_categories_parent_id ON categories(parent_id);
```

### 2.6 技能-分类关联 (skill_categories)

```sql
CREATE TABLE skill_categories (
    skill_id        BIGINT NOT NULL REFERENCES skills(id),
    category_id     BIGINT NOT NULL REFERENCES categories(id),
    PRIMARY KEY (skill_id, category_id)
);
```

### 2.7 订单 (orders)

```sql
CREATE TABLE orders (
    id              BIGSERIAL PRIMARY KEY,
    order_no        VARCHAR(64) NOT NULL UNIQUE,
    user_id         UUID NOT NULL,                   -- qooauth 用户 ID
    skill_id        BIGINT NOT NULL REFERENCES skills(id),
    version_id      BIGINT REFERENCES skill_versions(id),
    amount          DECIMAL(10,2) NOT NULL,
    currency        VARCHAR(3) DEFAULT 'USD',
    discount_amount DECIMAL(10,2) DEFAULT 0.00,
    tax_amount      DECIMAL(10,2) DEFAULT 0.00,
    total_amount    DECIMAL(10,2) NOT NULL,
    status          VARCHAR(16) DEFAULT 'pending',   -- pending/paid/completed/refunded/cancelled
    payment_method  VARCHAR(32),                     -- stripe/alipay/wechat
    payment_id      VARCHAR(255),                    -- 支付网关返回 ID
    paid_at         TIMESTAMP,
    completed_at    TIMESTAMP,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_orders_user_id ON orders(user_id);
CREATE INDEX idx_orders_order_no ON orders(order_no);
CREATE INDEX idx_orders_status ON orders(status);
```

### 2.8 License (licenses)

```sql
CREATE TABLE licenses (
    id              BIGSERIAL PRIMARY KEY,
    user_id         UUID NOT NULL,
    skill_id        BIGINT NOT NULL REFERENCES skills(id),
    version_id      BIGINT REFERENCES skill_versions(id),
    order_id        BIGINT REFERENCES orders(id),
    device_id       VARCHAR(128),                    -- 绑定的机器人设备 ID
    license_key     VARCHAR(255) NOT NULL UNIQUE,
    type            VARCHAR(16) DEFAULT 'perpetual', -- perpetual/subscription/trial
    status          VARCHAR(16) DEFAULT 'active',    -- active/expired/revoked
    starts_at       TIMESTAMP,
    expires_at      TIMESTAMP,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(user_id, skill_id, device_id)
);

CREATE INDEX idx_licenses_user_id ON licenses(user_id);
CREATE INDEX idx_licenses_device_id ON licenses(device_id);
CREATE INDEX idx_licenses_license_key ON licenses(license_key);
```

### 2.9 评价 (reviews)

```sql
CREATE TABLE reviews (
    id              BIGSERIAL PRIMARY KEY,
    skill_id        BIGINT NOT NULL REFERENCES skills(id),
    user_id         UUID NOT NULL,
    order_id        BIGINT REFERENCES orders(id),
    rating          SMALLINT NOT NULL CHECK (rating >= 1 AND rating <= 5),
    title           VARCHAR(255),
    content         TEXT,
    helpful_count   INTEGER DEFAULT 0,
    status          VARCHAR(16) DEFAULT 'published', -- published/hidden/removed
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(skill_id, user_id)                        -- 每用户每技能仅一条评价
);

CREATE INDEX idx_reviews_skill_id ON reviews(skill_id);
CREATE INDEX idx_reviews_user_id ON reviews(user_id);
```

### 2.10 技能统计 (skill_stats)

```sql
CREATE TABLE skill_stats (
    id              BIGSERIAL PRIMARY KEY,
    skill_id        BIGINT NOT NULL REFERENCES skills(id),
    date            DATE NOT NULL,
    downloads       INTEGER DEFAULT 0,
    active_users    INTEGER DEFAULT 0,
    revenue         DECIMAL(12,2) DEFAULT 0.00,
    crash_count     INTEGER DEFAULT 0,
    avg_rating      DECIMAL(3,2),
    review_count    INTEGER DEFAULT 0,
    UNIQUE(skill_id, date)
);

CREATE INDEX idx_skill_stats_skill_date ON skill_stats(skill_id, date);
```

### 2.11 开发者收益 (developer_revenue)

```sql
CREATE TABLE developer_revenue (
    id              BIGSERIAL PRIMARY KEY,
    developer_id    BIGINT NOT NULL REFERENCES developers(id),
    order_id        BIGINT NOT NULL REFERENCES orders(id),
    skill_id        BIGINT NOT NULL REFERENCES skills(id),
    gross_amount    DECIMAL(10,2) NOT NULL,           -- 用户支付金额
    platform_fee    DECIMAL(10,2) NOT NULL,           -- 平台分成
    developer_share DECIMAL(10,2) NOT NULL,           -- 开发者分成
    share_rate      DECIMAL(5,4) NOT NULL,            -- 分成比例 (0.7000 = 70%)
    currency        VARCHAR(3) DEFAULT 'USD',
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_developer_revenue_dev ON developer_revenue(developer_id);
CREATE INDEX idx_developer_revenue_date ON developer_revenue(created_at);
```

### 2.12 开发者结算 (developer_payouts)

```sql
CREATE TABLE developer_payouts (
    id              BIGSERIAL PRIMARY KEY,
    developer_id    BIGINT NOT NULL REFERENCES developers(id),
    amount          DECIMAL(12,2) NOT NULL,
    currency        VARCHAR(3) DEFAULT 'USD',
    period_start    DATE NOT NULL,
    period_end      DATE NOT NULL,
    status          VARCHAR(16) DEFAULT 'pending',    -- pending/processing/completed/failed
    payout_method   VARCHAR(32),
    transaction_id  VARCHAR(255),
    paid_at         TIMESTAMP,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_developer_payouts_dev ON developer_payouts(developer_id);
CREATE INDEX idx_developer_payouts_status ON developer_payouts(status);
```

### 2.13 设备技能安装记录 (device_skills)

```sql
CREATE TABLE device_skills (
    id              BIGSERIAL PRIMARY KEY,
    device_id       VARCHAR(128) NOT NULL,            -- 机器人 SN
    skill_id        BIGINT NOT NULL REFERENCES skills(id),
    version_id      BIGINT REFERENCES skill_versions(id),
    license_id      BIGINT REFERENCES licenses(id),
    status          VARCHAR(16) DEFAULT 'installing', -- installing/active/updating/uninstalling/removed
    installed_at    TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(device_id, skill_id)
);

CREATE INDEX idx_device_skills_device ON device_skills(device_id);
```

## 三、Redis 缓存设计

| Key 模式 | 类型 | TTL | 说明 |
|----------|------|-----|------|
| `store:skills:hot` | SortedSet | 1h | 热门技能排行 (score=下载量) |
| `store:skills:{skillId}:detail` | Hash | 30min | 技能详情缓存 |
| `store:categories:tree` | String(JSON) | 1h | 分类树结构 |
| `store:search:{query_hash}` | String(JSON) | 5min | 搜索结果缓存 |
| `store:featured` | String(JSON) | 15min | 精选推荐列表 |
| `store:rankings:{type}` | SortedSet | 30min | 排行榜 (free/paid/top_grossing/new) |
| `publisher:{userId}:skills` | Set | 30min | 开发者技能列表 |
| `order:{orderNo}:status` | String | 1h | 订单状态 |
| `session:{sessionId}` | Hash | 24h | 用户会话 |

## 五、RocketMQ 消息主题

| Topic | 生产者 | 消费者 | 说明 |
|-------|--------|--------|------|
| `store-skill-submitted` | API Gateway | 自动扫描服务 | 新技能提交触发扫描 |
| `store-scan-completed` | 自动扫描服务 | 审核服务 | 扫描完成后通知审核 |
| `store-skill-published` | 审核服务 | 搜索索引/缓存 | 技能发布更新索引 |
| `store-order-paid` | 支付服务 | License 服务/收益服务 | 支付成功后处理 |
| `store-license-created` | License 服务 | 端侧同步服务 | License 生成通知 |
| `store-stats-hourly` | 端侧上报 | 统计分析服务 | 使用统计汇总 |
| `store-crash-reported` | 端侧上报 | 崩溃分析服务 | 崩溃日志收集 |
