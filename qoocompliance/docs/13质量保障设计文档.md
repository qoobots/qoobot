### 2.1 测试示例

```java
@Test
void shouldGenerateEUChecklistForPersonalRobot() {
    ChecklistRequest req = ChecklistRequest.builder()
        .markets(List.of("EU"))
        .productType("personal_service_robot")
        .subprojects(List.of("qoobody", "qoobrain"))
        .build();
    
    Checklist checklist = checklistEngine.generate(req);
    
    assertThat(checklist.getTotalItems()).isGreaterThan(50);
    assertThat(checklist.getItems()).extracting("market")
        .containsOnly("EU");
    assertThat(checklist.getItems()).extracting("regulation")
        .contains("2006/42/EC", "GDPR", "2014/53/EU", "2011/65/EU");
}

@Test
void shouldDetectRegulationChange() {
    Regulation oldReg = loadRegulation("GDPR_2023.json");
    Regulation newReg = loadRegulation("GDPR_2024.json");
    
    List<Change> changes = regulatoryMonitor.detectChanges(oldReg, newReg);
    
    assertThat(changes).isNotEmpty();
    assertThat(changes).extracting(Change::getType)
        .contains("article_amended", "new_requirement");
}

@Test
void shouldRefuseUnauthorizedChecklistUpdate() {
    mockMvc.perform(put("/api/v1/checklists/CL-001/items/IT-001")
            .with(csrf())
            .header("Authorization", "Bearer " + engineerToken)
            .contentType(MediaType.APPLICATION_JSON)
            .content("{\"status\":\"passed\"}"))
        .andExpect(status().isForbidden()); // 非本人清单
}
```
