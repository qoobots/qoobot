## 2. API 设计

### 2.1 gRPC 服务定义

```protobuf
syntax = "proto3";
package brain_os;

// ============================================
// 认知服务 (Cognition Service)
// ============================================
service CognitionService {
  // 意图识别
  rpc RecognizeIntent(RecognizeIntentRequest) returns (IntentResponse);
  // 任务分解
  rpc DecomposeTask(DecomposeTaskRequest) returns (TaskDecompositionResponse);
  // 场景语义理解
  rpc UnderstandScene(SceneRequest) returns (SemanticContextResponse);
}

message RecognizeIntentRequest {
  string utterance = 1;          // 用户原始指令
  string session_id = 2;         // 会话 ID
  repeated string chat_history = 3; // 多轮对话历史
}

message IntentResponse {
  string intent_id = 1;
  string action = 2;             // fetch/place/navigate/inspect/...
  EntityRef target = 3;          // 目标物体
  EntityRef destination = 4;     // 目标位置 (可选)
  repeated Constraint constraints = 5;
  float confidence = 6;
  string explanation = 7;        // LLM 的解释
}

message DecomposeTaskRequest {
  string task_id = 1;
  IntentResponse intent = 2;
  string scene_id = 3;           // 当前场景快照
}

message TaskDecompositionResponse {
  string task_id = 1;
  repeated SubtaskInfo subtasks = 2;
  string raw_bt_json = 3;        // 可选: LLM 直接生成的行为树
}

message SubtaskInfo {
  string subtask_id = 1;
  int32 order = 2;
  string skill_name = 3;
  map<string, string> parameters = 4;
  repeated string depends_on = 5; // 依赖的 subtask_id
}

// ============================================
// 决策服务 (Decision Service)
// ============================================
service DecisionService {
  // 生成行为树和轨迹方案
  rpc GeneratePlan(GeneratePlanRequest) returns (PlanResponse);
  // 获取轨迹详情
  rpc GetTrajectory(GetTrajectoryRequest) returns (TrajectoryResponse);
  // 选择轨迹
  rpc SelectTrajectory(SelectTrajectoryRequest) returns (SelectTrajectoryResponse);
  // 执行行为树（流式返回进度）
  rpc ExecuteBehaviorTree(ExecuteBTRequest) returns (stream BTExecutionUpdate);
  // 暂停/恢复/取消执行
  rpc ControlExecution(ExecutionControlRequest) returns (ExecutionControlResponse);
}

message GeneratePlanRequest {
  string task_id = 1;
  string scene_id = 2;
  TaskDecompositionResponse decomposition = 3;
  CollaborationMode mode = 4;    // AUTONOMOUS/ADVISORY/CONFIRMATION
}

message PlanResponse {
  string plan_id = 1;
  string bt_json = 2;            // 行为树 JSON
  repeated TrajectoryInfo trajectories = 3;  // 候选轨迹
  CollaborationMode mode = 4;
}

message TrajectoryInfo {
  string trajectory_id = 1;
  StrategyType strategy = 2;     // OPTIMAL/CONSERVATIVE/...
  float score = 3;
  ScoreDetail score_detail = 4;  // 详细评分
  float duration = 5;            // 总时长(秒)
  repeated Waypoint waypoints = 6;
  repeated KeyEvent key_events = 7;
  ColorRGBA render_color = 8;    // 可视化颜色
}

message SelectTrajectoryRequest {
  string plan_id = 1;
  string trajectory_id = 2;
  DecisionSource source = 3;
  int32 decision_time_ms = 4;
  string human_id = 5;           // 可选
}

message BTExecutionUpdate {
  string plan_id = 1;
  string current_node_id = 2;
  string current_node_name = 3;
  NodeStatus status = 4;         // RUNNING/SUCCESS/FAILURE
  float progress = 5;            // 整体进度 0~1
  string message = 6;            // 人类可读的状态描述
}

// ============================================
// 控制服务 (Control Service)
// ============================================
service ControlService {
  // 执行运动轨迹
  rpc ExecuteMotion(ExecuteMotionRequest) returns (stream MotionUpdate);
  // 直接控制（覆写模式）
  rpc DirectControl(stream DirectCommand) returns (stream JointState);
  // 紧急停止
  rpc EmergencyStop(EmergencyStopRequest) returns (EmergencyStopResponse);
  // 查询机器人状态
  rpc GetRobotState(GetRobotStateRequest) returns (RobotStateResponse);
}

message ExecuteMotionRequest {
  string motion_id = 1;
  string trajectory_id = 2;
  double speed_scale = 3;        // 速度缩放 0~1
}

message MotionUpdate {
  string motion_id = 1;
  MotionStatus status = 2;
  float progress = 3;
  JointState current_state = 4;
  Pose6D current_ee_pose = 5;
}

message DirectCommand {
  oneof command {
    JointVelocity joint_vel = 1;  // 关节速度控制
    Twist ee_twist = 2;          // 末端速度控制
    GripperCommand gripper = 3;   // 手爪控制
  }
}

// ============================================
// 感知服务 (Perception Service)
// ============================================
service PerceptionService {
  // 获取当前场景快照
  rpc GetCurrentScene(GetSceneRequest) returns (SceneResponse);
  // 场景流（持续推送）
  rpc StreamScene(GetSceneRequest) returns (stream SceneUpdate);
  // 查询特定物体
  rpc QueryObject(QueryObjectRequest) returns (ObjectResponse);
}

message SceneResponse {
  string scene_id = 1;
  Timestamp timestamp = 2;
  Pose6D robot_pose = 3;
  repeated DetectedObject objects = 4;
  OccupancyGridInfo occupancy = 5;
  string gs_model_ref = 6;       // 3DGS 模型引用
}

message SceneUpdate {
  string scene_id = 1;
  UpdateType update_type = 2;    // FULL/POSE_ONLY/NEW_OBJECT/OBJECT_LOST
  SceneResponse full_scene = 3;  // 仅 FULL 时有值
  Pose6D new_pose = 4;
  DetectedObject new_object = 5;
  string lost_object_id = 6;
}

// ============================================
// 知识服务 (Knowledge Service)
// ============================================
service KnowledgeService {
  // 语义检索
  rpc SemanticSearch(SemanticSearchRequest) returns (SemanticSearchResponse);
  // 记录经验
  rpc RecordEpisode(EpisodeRecord) returns (RecordResponse);
  // 查询历史任务
  rpc QueryHistory(HistoryQueryRequest) returns (HistoryQueryResponse);
}

message SemanticSearchRequest {
  string query = 1;
  int32 top_k = 2;
  KnowledgeCategory category_filter = 3;  // 可选
}

message SemanticSearchResponse {
  repeated KnowledgeResult results = 1;
  float search_time_ms = 2;
}

message KnowledgeResult {
  string knowledge_id = 1;
  string category = 2;
  string content = 3;
  float similarity = 4;
}

// ============================================
// 安全服务 (Safety Service)
// ============================================
service SafetyService {
  // 获取安全状态
  rpc GetSafetyStatus(GetSafetyStatusRequest) returns (SafetyStatusResponse);
  // 安全事件流
  rpc StreamSafetyEvents(GetSafetyStatusRequest) returns (stream SafetyEvent);
}

// ============================================
// 通用类型
// ============================================

message Pose6D {
  double x = 1; double y = 2; double z = 3;
  double qx = 4; double qy = 5; double qz = 6; double qw = 7;
}

message Point3D {
  double x = 1; double y = 2; double z = 3;
}

message BBox3D {
  Point3D center = 1;
  Point3D size = 2;
}

message EntityRef {
  string entity_type = 1;
  string entity_id = 2;
  map<string, string> attributes = 3;
}

message Constraint {
  string type = 1;      // avoid/speed_limit/force_limit/priority/...
  string target = 2;
  string value = 3;
  int32 priority = 4;
}

message ScoreDetail {
  float safety_score = 1;
  float efficiency_score = 2;
  float smoothness_score = 3;
  float success_probability = 4;
}

message Waypoint {
  float timestamp_sec = 1;
  repeated float joint_positions = 2;
  Pose6D ee_pose = 3;
  float gripper_state = 4;
}

message KeyEvent {
  float event_time = 1;
  string event_type = 2;
  string description = 3;
}

message ColorRGBA { float r = 1; float g = 2; float b = 3; float a = 4; }

enum StrategyType {
  STRATEGY_UNKNOWN = 0;
  OPTIMAL = 1;
  CONSERVATIVE = 2;
  AGGRESSIVE = 3;
  EXPLORATORY = 4;
  ADVERSARIAL = 5;
}

enum CollaborationMode {
  AUTONOMOUS = 0;
  ADVISORY = 1;
  CONFIRMATION = 2;
}

enum DecisionSource {
  SOURCE_UNKNOWN = 0;
  AUTO = 1;
  HUMAN = 2;
  OVERRIDE = 3;
}
```

### 2.2 Python SDK API

```python
# brain_os SDK — 开发者使用示例

import brain_os
from brain_os.types import Pose6D, StrategyType, CollaborationMode

# 1. 连接
robot = brain_os.connect(host="localhost:50051")

# 2. 感知
scene = robot.perception.get_current_scene()
objects = robot.perception.query_objects(class_name="cup", color="red")
robot.perception.stream_scene(callback=lambda update: print(update))

# 3. 认知
intent = robot.cognition.recognize("把桌上的红色杯子递给我")
task = robot.cognition.decompose_task(intent, scene.id)

# 4. 决策
plan = robot.decision.generate_plan(
    task=task,
    scene_id=scene.id,
    mode=CollaborationMode.ADVISORY
)

# 5. 人在回路 (SDK 可编程选择)
selected = plan.trajectories[0]  # 最优策略
if plan.trajectories[1].score > 0.85:
    selected = plan.trajectories[1]  # SDK 逻辑选择保守策略

robot.decision.select_trajectory(plan.id, selected.id)

# 6. 执行（异步流式）
async for update in robot.decision.execute_bt(plan.id):
    print(f"Progress: {update.progress:.0%} | {update.message}")

# 7. 知识
results = robot.knowledge.semantic_search("如何安全地抓取玻璃杯")
for r in results:
    print(f"[{r.similarity:.2f}] {r.content}")

# 8. 直接控制
robot.control.move_to(Pose6D(x=0.5, y=0.3, z=0.8))
robot.control.gripper_open()
robot.control.gripper_close(force=0.5)

# 9. 语音交互
robot.speak("任务完成")
```

## 4. WebSocket 事件流（brain_viz ↔ brain_ai）

### 4.1 事件通道

```typescript
// 前端 WebSocket 事件类型

interface BrainEvent {
  event_type: string;
  timestamp: number;
  payload: any;
}

// 轨迹推送
interface TrajectoryPushEvent extends BrainEvent {
  event_type: "trajectories_planned";
  payload: {
    plan_id: string;
    trajectories: TrajectoryInfo[];
    mode: CollaborationMode;
    timeout_ms: number;  // 超时时间
  };
}

// 执行进度
interface ExecutionProgressEvent extends BrainEvent {
  event_type: "execution_progress";
  payload: {
    plan_id: string;
    current_node: string;
    progress: number;
    status: string;
    message: string;
  };
}

// 安全告警
interface SafetyAlertEvent extends BrainEvent {
  event_type: "safety_alert";
  payload: {
    incident_id: string;
    level: "S0" | "S1" | "S2" | "S3";
    type: string;
    description: string;
  };
}

// 场景更新
interface SceneUpdateEvent extends BrainEvent {
  event_type: "scene_update";
  payload: {
    scene_id: string;
    objects: DetectedObject[];
    gs_model_ref: string;
  };
}
```

## 5. 通信协议选型

| 通信场景 | 协议 | 序列化 | 理由 |
|----------|------|--------|------|
| **内部节点通信** | ROS 2 Topic/Service | ROS 2 原生 msg | 低延迟、生态集成 |
| **AI ↔ Core 通信** | gRPC (unary + streaming) | Protobuf | 跨语言、强类型、双向流 |
| **Core ↔ 实时控制** | ROS 2 Action | ROS 2 原生 | 可中断、反馈机制 |
| **Viz ↔ AI 事件** | gRPC + WebSocket | Protobuf + JSON | 实时推送、Web 友好 |
| **SDK ↔ Brain OS** | gRPC (unary + streaming) | Protobuf | 类型安全、跨语言、高性能 |
| **云端模型调用** | gRPC / REST | Protobuf / JSON | 兼容现有 API |
